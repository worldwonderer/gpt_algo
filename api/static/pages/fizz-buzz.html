Fizz Buzz 标签: 数学 字符串 模拟    难度: Easy

Submission


def fizzBuzz(n):
    result = []
    for i in range(1, n+1):
        if i % 3 == 0 and i % 5 == 0:
            result.append("FizzBuzz")
        elif i % 3 == 0:
            result.append("Fizz")
        elif i % 5 == 0:
            result.append("Buzz")
        else:
            result.append(str(i))
    return result


class Solution:
    def fizzBuzz(self, n: int) -> List[str]:
        result = fizzBuzz(n)
        return result

运行时间: 20 ms

内存: 17.4 MB

Explain

此题解采用了直接遍历的方法。对于给定的整数n，我们从1遍历到n，对于每个整数i，我们检查它是否能被3和5同时整除、只能被3整除、只能被5整除，或者都不能整除。根据这些条件，我们将相应的字符串（'FizzBuzz'、'Fizz'、'Buzz'或者数字本身的字符串形式）添加到结果数组中。

时间复杂度: O(n)

空间复杂度: O(n)

def fizzBuzz(n):
    result = []  # 结果数组
    for i in range(1, n+1):  # 遍历1到n
        if i % 3 == 0 and i % 5 == 0:  # 同时是3和5的倍数
            result.append('FizzBuzz')
        elif i % 3 == 0:  # 是3的倍数
            result.append('Fizz')
        elif i % 5 == 0:  # 是5的倍数
            result.append('Buzz')
        else:  # 都不是
            result.append(str(i))  # 转换为字符串添加到结果数组
    return result

class Solution:
    def fizzBuzz(self, n: int) -> List[str]:
        result = fizzBuzz(n)  # 调用fizzBuzz函数
        return result
