Explain

该题解首先通过 strip 去除原字符串两端的空格并分割成单词列表。然后计算总空格数，通过总长度减去所有单词长度之和得到。如果单词数量大于1，计算相邻单词间的平均空格数，并确定末尾的剩余空格数。使用 join 结合这些空格重新组合单词。如果只有一个单词，直接将所有空格放在单词后面。

时间复杂度: O(n)

空间复杂度: O(n)

class Solution:
    def reorderSpaces(self, text: str) -> str:
        textStrip = text.strip()  # 去除首尾空格
        textArray = textStrip.split()  # 分割单词
        spaceNumSum = len(text) - len(''.join(textArray))  # 计算总空格数
        if len(textArray) > 1:  # 如果单词数量大于1
            spaceNum = int(spaceNumSum / (len(textArray) - 1))  # 计算单词间平均空格数
            lastSpace = spaceNumSum - (len(textArray) - 1) * spaceNum  # 计算末尾空格数
            text = ''.join((' ') * spaceNum).join(textArray) + ' ' * lastSpace  # 重构字符串
            return text
        else:  # 如果只有一个单词
            text = textArray[0] + ' ' * spaceNumSum  # 将所有空格放在单词后面
            return text
