class Solution:
    def licenseKeyFormatting(self, s: str, k: int) -> str:

        s = ''.join(s.split('-'))
        n = len(s)
        if n % k != 0:
            first = n % k  
        else:
            first = k
        res = str.upper(s[:first])
        for i in range(first, n, k):
            res += '-' + str.upper(s[i : i + k])
        return res

运行时间: 22 ms

内存: 16.9 MB

Explain

这个题解的思路是先将原字符串中的破折号去掉，得到一个只包含字母和数字的字符串。然后计算出第一组的长度，可能会比k短。接着将字符串的第一组转换为大写，作为结果字符串的开头。从第一组之后，每k个字符一组，在它们之前加上破折号，并转换为大写，拼接到结果字符串后面。最终返回重新格式化后的许可密钥字符串。

时间复杂度: O(n)

空间复杂度: O(n)

class Solution:
    def licenseKeyFormatting(self, s: str, k: int) -> str:
        # 去掉字符串中的破折号
        s = ''.join(s.split('-'))
        n = len(s)
        
        # 计算第一组的长度
        if n % k != 0:
            first = n % k  
        else:
            first = k
        
        # 将第一组转换为大写，作为结果字符串的开头
        res = str.upper(s[:first])
        
        # 从第一组之后，每k个字符一组，转换为大写并加上破折号，拼接到结果字符串
        for i in range(first, n, k):
            res += '-' + str.upper(s[i : i + k])
        
        return res
