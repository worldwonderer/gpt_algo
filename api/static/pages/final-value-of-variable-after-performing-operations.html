from typing import List

class Solution:
    def finalValueAfterOperations(self, operations: List[str]) -> int:
        X = 0  # 初始化变量 X 的值为 0
        
        for operation in operations:
            if operation == "++X" or operation == "X++":
                X += 1  # 执行加 1 操作
            elif operation == "--X" or operation == "X--":
                X -= 1  # 执行减 1 操作
        
        return X  # 返回最终值

solution = Solution()
operations = ["--X", "X++", "X++"]
result = solution.finalValueAfterOperations(operations)
print(result)  # 输出: 1

运行时间: 18 ms

内存: 16.1 MB

Explain

该题解通过迭代输入的操作列表来更新变量X的值。对于每个操作，根据其具体是增加还是减少操作，决定是对X进行加1或减1的操作。最终返回经过所有操作后的X的值。

时间复杂度: O(n)

空间复杂度: O(1)
