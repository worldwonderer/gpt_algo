class Solution:
    def calPoints(self, operations: List[str]) -> int:
        score_sum = 0
        tmp_op = []
        for i in range(len(operations)):
            ele = operations[i]
            if ele == "C" :
                score_sum -= tmp_op[-1]
                tmp_op.pop()
            elif ele == "D":
                var = 2 * tmp_op[-1]
                score_sum += var
                tmp_op.append(var)
            elif ele == "+":
                var = tmp_op[-1] + tmp_op[-2]
                score_sum += var
                tmp_op.append(var)
            else:
                score_sum += int(ele)
                tmp_op.append(int(ele))
        return score_sum

运行时间: 12 ms

内存: 16.2 MB

Explain

该题解使用一个栈（列表）来模拟整个计分过程。遍历给定的操作列表，根据不同的操作符执行相应的计分操作。对于整数，直接将其加入到总分中，并将其压入栈中；对于 'C' 操作，从总分中减去栈顶元素，并将栈顶元素弹出；对于 'D' 操作，将栈顶元素的两倍加入到总分中，并将其压入栈中；对于 '+' 操作，将栈顶两个元素的和加入到总分中，并将其压入栈中。最后返回总分即可。

时间复杂度: O(n)

空间复杂度: O(n)

class Solution:
    def calPoints(self, operations: List[str]) -> int:
        score_sum = 0  # 总分
        tmp_op = []  # 栈，用于存储计分过程中的中间结果
        
        for i in range(len(operations)):
            ele = operations[i]
            
            if ele == "C" :
                # 对于 'C' 操作，从总分中减去栈顶元素，并将栈顶元素弹出
                score_sum -= tmp_op[-1]
                tmp_op.pop()
            elif ele == "D":
                # 对于 'D' 操作，将栈顶元素的两倍加入到总分中，并将其压入栈中
                var = 2 * tmp_op[-1]
                score_sum += var
                tmp_op.append(var)
            elif ele == "+":
                # 对于 '+' 操作，将栈顶两个元素的和加入到总分中，并将其压入栈中
                var = tmp_op[-1] + tmp_op[-2]
                score_sum += var
                tmp_op.append(var)
            else:
                # 对于整数，直接将其加入到总分中，并将其压入栈中
                score_sum += int(ele)
                tmp_op.append(int(ele))
        
        return score_sum
