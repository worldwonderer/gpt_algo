Explain

这个题解采用了模拟二进制加法的方式来计算两个二进制字符串的和。从两个字符串的末尾开始，逐位相加，同时记录进位。最后将计算结果逆序，得到最终的二进制字符串。

时间复杂度: O(n)

空间复杂度: O(1)

class Solution:
    def addBinary(self, a: str, b: str) -> str:
        ans = ""
        i = len(a) - 1
        j = len(b) - 1
        carry = 0
        
        # 从末尾开始，逐位相加
        while i >= 0 or j >= 0 or carry == 1:
            # 获取当前位的值，如果已经越界则为0
            v1 = int(a[i]) if i >= 0 else 0
            v2 = int(b[j]) if j >= 0 else 0

            # 计算当前位的和，包括来自低位的进位
            total = v1 + v2 + carry
            
            # 判断当前位是否需要进位
            if total >= 2:
                total -= 2
                carry = 1
            else:
                carry = 0
            
            # 将当前位的结果添加到最终答案的开头
            ans = str(total) + ans
            
            # 移动到下一位
            i -= 1
            j -= 1
        
        return ans
