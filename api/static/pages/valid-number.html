class Solution:
    def isNumber(self, s: str) -> bool:
        split_i = -1
        for i, c in enumerate(s):
            if c == "E" or c == "e":
                if split_i != -1:
                    return False
                split_i = i
        length = len(s)
        if split_i == -1:
            return self.check(s, 0, length-1, False)
        else:
            return self.check(s, 0, split_i-1, False) and self.check(s, split_i+1, length-1, True)

    def check(self, s, start, end, mustInteger):
        if start > end:
            return False
        if s[start] == "+" or s[start] == "-":
            start += 1
        has_dot = False
        has_num = False
        for i in range(start, end+1):
            if s[i] == '.':
                if has_dot or mustInteger:
                    return False
                has_dot = True
            elif s[i] >= "0" and s[i] <= "9":
                has_num = True
            else:
                return False
        return has_num

运行时间: 44 ms

内存: 15.1 MB

Explain

该题解采用分治的思路来判断字符串是否为有效数字。首先遍历字符串找到 'e' 或 'E'，如果存在则将字符串分成两部分，分别判断 'e' 前面的部分是否为有效的小数或整数，'e' 后面的部分是否为有效的整数。如果不存在 'e'，则直接判断整个字符串是否为有效的小数或整数。check 函数用于判断字符串的一部分是否满足有效小数或整数的条件。

时间复杂度: O(n)

空间复杂度: O(1)

class Solution:
    def isNumber(self, s: str) -> bool:
        split_i = -1
        for i, c in enumerate(s):
            if c == "E" or c == "e":
                if split_i != -1:
                    return False
                split_i = i
        length = len(s)
        if split_i == -1:
            # 如果不存在 'e'，直接判断整个字符串
            return self.check(s, 0, length-1, False) 
        else:
            # 如果存在 'e'，分别判断 'e' 两侧的字符串
            return self.check(s, 0, split_i-1, False) and self.check(s, split_i+1, length-1, True)

    def check(self, s, start, end, mustInteger):
        if start > end:
            return False
        if s[start] == "+" or s[start] == "-":
            start += 1
        has_dot = False
        has_num = False
        for i in range(start, end+1):
            if s[i] == '.':
                if has_dot or mustInteger:
                    return False
                has_dot = True
            elif s[i] >= "0" and s[i] <= "9":
                has_num = True
            else:
                return False
        # 必须至少有一个数字
        return has_num
