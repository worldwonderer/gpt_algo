class FoodRatings:
    def __init__(self, foods: List[str], cuisines: List[str], ratings: List[int]):
        self.fs = {}
        self.cs = defaultdict(list)
        for f, c, r in zip(foods, cuisines, ratings):
            self.fs[f] = [r, c]
            heappush(self.cs[c], (-r, f))

    def changeRating(self, food: str, newRating: int) -> None:
        f = self.fs[food]
        heappush(self.cs[f[1]], (-newRating, food))  # 直接添加新数据，后面 highestRated 再删除旧的
        f[0] = newRating

    def highestRated(self, cuisine: str) -> str:
        h = self.cs[cuisine]
        while -h[0][0] != self.fs[h[0][1]][0]:  # 堆顶的食物评分不等于其实际值
            heappop(h)
        return h[0][1]

运行时间: 252 ms

内存: 49.9 MB

Explain

这个题解采用了哈希表和最小堆的数据结构来有效管理食物的评分和查询。具体来说，使用一个哈希表 `fs` 存储每个食物的评分和所属烹饪方式，使用 `cs` 字典的每个键对应一个最小堆，存储每种烹饪方式下所有食物的评分（存储为负数以模拟最大堆）和食物名。当修改食物评分时，直接在对应烹饪方式的堆中添加新的评分，旧的评分将在执行 `highestRated` 操作时被检查并移除，保证了堆中总是有最新的评分数据。这种做法虽然在更改评分时较快，但可能导致堆中有多余的元素，需要在查询最高评分时进行清理。

时间复杂度: O(log n)

空间复杂度: O(n)

class FoodRatings:
    def __init__(self, foods: List[str], cuisines: List[str], ratings: List[int]):
        self.fs = {}  # 存储食物及其评分和烹饪方式
        self.cs = defaultdict(list)  # 存储每种烹饪方式的食物评分堆
        for f, c, r in zip(foods, cuisines, ratings):
            self.fs[f] = [r, c]
            heappush(self.cs[c], (-r, f))  # 使用负值模拟最大堆

    def changeRating(self, food: str, newRating: int) -> None:
        f = self.fs[food]
        heappush(self.cs[f[1]], (-newRating, food))  # 将新评分添加到堆中
        f[0] = newRating  # 更新哈希表中的评分

    def highestRated(self, cuisine: str) -> str:
        h = self.cs[cuisine]
        while -h[0][0] != self.fs[h[0][1]][0]:  # 清理堆顶的无效评分
            heappop(h)
        return h[0][1]  # 返回当前最高评分的食物名
