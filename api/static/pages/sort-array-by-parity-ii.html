class Solution:
    def sortArrayByParityII(self, nums: List[int]) -> List[int]:
        a=[]
        b=[]
        d=[]
        for i in nums:
            if i%2==0:
                a.append(i)
            else:

        for m in range(len(a)):
            d.extend([a[m],b[m]])
        return d

Explain

这个题解的思路是首先遍历原数组，将奇数和偶数分别存储在两个列表中。然后，按照题目要求，依次从两个列表中取出一个奇数和一个偶数，交替放入新的列表中，最后返回这个新列表。

时间复杂度: O(n)

空间复杂度: O(n)

class Solution:
    def sortArrayByParityII(self, nums: List[int]) -> List[int]:
        a = []  # 存储偶数
        b = []  # 存储奇数
        d = []  # 存储最终结果
        for i in nums:
            if i % 2 == 0:
                a.append(i)
            else:
                b.append(i)
        for m in range(len(a)):
            d.extend([a[m], b[m]])  # 交替添加奇数和偶数到结果列表
        return d
