class Solution:
    def findLonelyPixel(self, picture: List[List[str]]) -> int:
        m, n = len(picture), len(picture[0])
        cntRow, cntCol = [0]*m, [0]*n
        for i in range(m):
            for j in range(n):
                if picture[i][j] == 'B':
                    cntRow[i] += 1
                    cntCol[j] += 1
        #print(cntRow, cntCol)
        res = 0
        for i in range(m):
            for j in range(n):
                if picture[i][j] == 'B' and cntRow[i] == 1 and cntCol[j] == 1:
                    res += 1
        return res

运行时间: 44 ms

内存: 21.6 MB

Explain

这个题解的思路是先遍历整个二维数组，统计每一行和每一列中 'B' 的个数，记录在 cntRow 和 cntCol 两个数组中。然后再次遍历二维数组，对于每个 'B'，如果它所在的行和列的 'B' 的个数都为 1，说明这个 'B' 是孤独像素，结果加 1。最后返回孤独像素的总数。

时间复杂度: O(mn)

空间复杂度: O(m+n)

class Solution:
    def findLonelyPixel(self, picture: List[List[str]]) -> int:
        m, n = len(picture), len(picture[0])
        cntRow, cntCol = [0]*m, [0]*n
        # 遍历二维数组，统计每行每列的 'B' 的个数
        for i in range(m):
            for j in range(n):
                if picture[i][j] == 'B':
                    cntRow[i] += 1
                    cntCol[j] += 1
        
        res = 0
        # 再次遍历二维数组，对每个 'B' 判断它是否是孤独像素
        for i in range(m):
            for j in range(n):
                if picture[i][j] == 'B' and cntRow[i] == 1 and cntCol[j] == 1:
                    res += 1
        return res
