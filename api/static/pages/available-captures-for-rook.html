class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        row, col = '', ''

        for i in board:
            row += ''.join(i)
            row += ','
        
        for j in range(8):
            for k in range(8):
                col += board[k][j]
            col += ','

        s = row + col
        s = s.replace('.', '')

        return s.count('pR') + s.count('Rp')

运行时间: 25 ms

内存: 16.1 MB

Explain

该题解的主要思路是首先将棋盘的行和列转化为字符串形式，然后组合到一起，形成一个单一的长字符串。接着，通过删除所有表示空格的'.'字符，使得字符串中只剩下棋子的字符。这样，'pR'和'Rp'将直接表示车（Rook）能够捕获的黑卒（pawn）的情况，即这两种组合分别代表黑卒在车的左侧或右侧，上侧或下侧。最后，统计这两种组合在字符串中出现的次数之和，即为答案。

时间复杂度: O(1)

空间复杂度: O(1)

class Solution:
    def numRookCaptures(self, board: List[List[str]]) -> int:
        row, col = '', ''

        # 构建行信息的字符串，每行之后添加一个分隔符','
        for i in board:
            row += ''.join(i)
            row += ','
        
        # 构建列信息的字符串，每列之后添加一个分隔符','
        for j in range(8):
            for k in range(8):
                col += board[k][j]
            col += ','

        # 合并行和列信息，形成单一字符串，并移除所有空格字符
        s = row + col
        s = s.replace('.', '')

        # 计算可以直接捕获的黑卒的数量
        return s.count('pR') + s.count('Rp')
