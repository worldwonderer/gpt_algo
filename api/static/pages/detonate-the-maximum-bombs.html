class Solution:
    def maximumDetonation(self, bombs: List[List[int]]) -> int:
        ##给定一个有向图（可能存在环），求目标起点，使得能够到达的节点数目最多
        n=len(bombs)
        nxt=[[] for _ in range(n)]
        for i in range(n):
            x1,y1,r1=bombs[i]
            for j in range(i):
                x2,y2,r2=bombs[j]
                d2=(x1-x2)**2+(y1-y2)**2
                if d2<=r1**2:
                    nxt[i].append(j)
                if d2<=r2**2:
                    nxt[j].append(i)
                
        def forward(x,vis):
            vis.add(x)
            for y in nxt[x]:
                if y not in vis:
                    forward(y,vis)
        ans=1
        vis=set()
        for i in range(n):
            if i in vis:continue
            newVis=set()
            forward(i,newVis)
            ans=max(ans,len(newVis))
            vis |=newVis
        return ans

Explain

这道题目可以视为图的问题，其中每个炸弹是一个节点，如果一个炸弹可以引爆另一个炸弹，则在这两个炸弹之间存在一条有向边。首先，我们通过计算两个炸弹之间的距离与半径比较，来构建这个有向图的邻接表。接着，从每个节点开始，使用深度优先搜索（DFS）来探索从该节点出发能够到达多少其他节点，即引爆多少其他炸弹。最后，返回能够引爆最多炸弹的数量。

时间复杂度: O(n^2)

空间复杂度: O(n^2)

class Solution:
    def maximumDetonation(self, bombs: List[List[int]]) -> int:
        n = len(bombs)
        nxt = [[] for _ in range(n)]
        for i in range(n):
            x1, y1, r1 = bombs[i]
            for j in range(n):
                if i != j:
                    x2, y2, r2 = bombs[j]
                    d2 = (x1 - x2)**2 + (y1 - y2)**2
                    if d2 <= r1**2:
                        nxt[i].append(j)

        def forward(x, vis):
            vis.add(x)
            for y in nxt[x]:
                if y not in vis:
                    forward(y, vis)

        ans = 1
        vis = set()
        for i in range(n):
            if i in vis:
                continue
            newVis = set()
            forward(i, newVis)
            ans = max(ans, len(newVis))
            vis |= newVis
        return ans
