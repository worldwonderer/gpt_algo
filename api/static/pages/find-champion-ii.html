class Solution:
    def findChampion(self, n: int, edges: List[List[int]]) -> int:
        champion = list(range(n))
        for i in edges:
            champion[i[1]] = -1
        ans = []
        for i in champion:
            if i > -1:
                ans.append(i)
        return ans[0] if len(ans) == 1 else -1

运行时间: 51 ms

内存: 17.0 MB

Explain

题解通过维护一个包含所有队伍的列表来标记可能的冠军队伍。遍历给定的边，对于每一条边 [u, v]，标记 v 队因为存在比它强的 u 队而不可能是冠军，通过将 champion[v] 设为 -1 实现。最后，所有未被标记为 -1 的队伍即是没有比它强的队伍的候选者。如果存在唯一的候选者，则这个候选者是冠军；如果不存在或存在多个，则返回 -1。

时间复杂度: O(n + m)

空间复杂度: O(n)

class Solution:
    def findChampion(self, n: int, edges: List[List[int]]) -> int:
        # 初始化一个列表，包含所有队伍
        champion = list(range(n))
        # 标记每个非冠军队伍，即存在强队的队伍
        for i in edges:
            champion[i[1]] = -1
        # 筛选可能的冠军队伍
        ans = []
        for i in champion:
            if i > -1:
                ans.append(i)
        # 判断是否有唯一冠军
        return ans[0] if len(ans) == 1 else -1
