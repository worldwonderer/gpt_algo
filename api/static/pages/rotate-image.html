class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """
        # 转置
        n = len(matrix)
        for i in range(n):
            for j in range(i, n):
                matrix[i][j], matrix[j][i] = matrix[j][i], matrix[i][j]

        # 每一行反转
        for i in range(n):
            matrix[i] = matrix[i][::-1]

运行时间: 44 ms

内存: 14.8 MB

Explain

这个题解采用了两步操作来实现矩阵的顺时针旋转90度。第一步是将矩阵进行转置，即将矩阵的行列进行互换。第二步是将转置后的矩阵的每一行进行反转。经过这两步操作，原矩阵就被顺时针旋转了90度。这种方法可以在原地修改矩阵，不需要使用额外的矩阵空间。

时间复杂度: O(n^2)

空间复杂度: O(1)

class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """
        n = len(matrix)
        
        # 第一步：转置矩阵
        for i in range(n):
            for j in range(i, n):
                matrix[i][j], matrix[j][i] = matrix[j][i], matrix[i][j]
        
        # 第二步：反转每一行
        for i in range(n):
            matrix[i] = matrix[i][::-1]
