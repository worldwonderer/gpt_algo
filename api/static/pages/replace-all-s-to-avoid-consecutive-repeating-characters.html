class Solution:

        res = list(s)
        n = len(res)
        for i in range(n):
            if res[i] == '?':
                for b in "abc":

                        res[i] = b
                        break

Explain

这个题解的思路是遍历给定的字符串，每次遇到 '?' 字符时，尝试用 'a', 'b', 'c' 中的一个字母替换，确保替换后的字符不与前后字符相同。这样可以保证结果字符串中没有连续重复的字符。

时间复杂度: O(n)

空间复杂度: O(n)

class Solution:
    def modifyString(self, s: str) -> str:
        res = list(s)  # 将字符串转换为列表
        n = len(res)  # 获取字符串的长度
        for i in range(n):  # 遍历字符串
            if res[i] == '?':  # 如果遇到 '?' 字符
                for b in "abc":  # 尝试用 'a', 'b', 'c' 中的一个字母替换
                    if not (i > 0 and res[i - 1] == b or i < n - 1 and res[i + 1] == b):  # 确保替换后的字符不与前后字符相同
                        res[i] = b  # 进行替换
                        break
        return ''.join(res)  # 将列表转换回字符串并返回
